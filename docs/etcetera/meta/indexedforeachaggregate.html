template IndexedForeachAggregate(T)

Template defining a delegate suitable to be used as an indexed ForeachAggregate.

Parameters

T The type of the foreach aggregate.

See Also

http://dlang.org/statement.html#ForeachStatement http://ddili.org/ders/d.en/foreach_opapply.html

Examples

class Foo
{
	private string[] _data = ["foo", "bar", "baz", "qux"];

	public int opApply(IndexedForeachAggregate!(string) dg)
	{
		int result;
		size_t index;

		for (int x = 0; x < this._data.length; index++, x++)
		{
			result = dg(index, this._data[x]);

			if (result)
			{
				break;
			}
		}

		return result;
	}
}

auto foo = new Foo();

foreach (index, value; foo)
{
	// ...
}
